REDUCE<br>
다음과 같이 사용 배열.reduce((누적값, 현재값, 인덱스, 요소) => {return 결과}, 누적값의 초기값);
ex)<br>
const oneTwoThree = [1, 2, 3];<br>
let result = oneTwoThree.reduce((acc, cur, i) => { <br>
  console.log(acc, cur, i); <br>
  return acc + cur; <br>
}, 0); <br>
// 0 1 0 <br>
// 1 2 1 <br>
// 3 3 2 <br>
result; //6
